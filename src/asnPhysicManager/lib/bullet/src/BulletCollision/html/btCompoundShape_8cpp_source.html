00061 void btCompoundShape::getAabb(const btTransform& trans,btVector3& aabbMin,btVector3& aabbMax) const
00062 {
00063         btVector3 localHalfExtents = btScalar(0.5)*(m_localAabbMax-m_localAabbMin);
00064         btVector3 localCenter = btScalar(0.5)*(m_localAabbMax+m_localAabbMin);
00065         
00066         btMatrix3x3 abs_b = trans.getBasis().absolute();  
00067 
00068         btPoint3 center = trans(localCenter);
00069 
00070         btVector3 extent = btVector3(abs_b[0].dot(localHalfExtents),
00071                    abs_b[1].dot(localHalfExtents),
00072                   abs_b[2].dot(localHalfExtents));
00073         extent += btVector3(getMargin(),getMargin(),getMargin());
00074 
00075         aabbMin = center - extent;
00076         aabbMax = center + extent;
