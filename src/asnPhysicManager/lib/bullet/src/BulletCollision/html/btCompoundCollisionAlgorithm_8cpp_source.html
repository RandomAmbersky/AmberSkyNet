00054 void btCompoundCollisionAlgorithm::processCollision (btCollisionObject* body0,btCollisionObject* body1,const btDispatcherInfo& dispatchInfo,btManifoldResult* resultOut)
00055 {
00056         btCollisionObject* colObj = m_isSwapped? body1 : body0;
00057         btCollisionObject* otherObj = m_isSwapped? body0 : body1;
00058 
00059         assert (colObj->getCollisionShape()->isCompound());
00060         btCompoundShape* compoundShape = static_cast<btCompoundShape*>(colObj->getCollisionShape());
00061 
00062         //We will use the OptimizedBVH, AABB tree to cull potential child-overlaps
00063         //If both proxies are Compound, we will deal with that directly, by performing sequential/parallel tree traversals
00064         //given Proxy0 and Proxy1, if both have a tree, Tree0 and Tree1, this means:
00065         //determine overlapping nodes of Proxy1 using Proxy0 AABB against Tree1
00066         //then use each overlapping node AABB against Tree0
00067         //and vise versa.
00068 
00069         int numChildren = m_childCollisionAlgorithms.size();
00070         int i;
00071         for (i=0;i<numChildren;i++)
00072         {
00073                 //temporarily exchange parent btCollisionShape with childShape, and recurse
00074                 btCollisionShape* childShape = compoundShape->getChildShape(i);
00075 
00076                 //backup
00077                 btTransform     orgTrans = colObj->getWorldTransform();
00078                 btCollisionShape* orgShape = colObj->getCollisionShape();
00079 
00080                 const btTransform& childTrans = compoundShape->getChildTransform(i);
00081                 //btTransform   newChildWorldTrans = orgTrans*childTrans ;
00082                 colObj->setWorldTransform( orgTrans*childTrans );
00083                 //the contactpoint is still projected back using the original inverted worldtrans
00084                 colObj->setCollisionShape( childShape );
00085                 m_childCollisionAlgorithms[i]->processCollision(colObj,otherObj,dispatchInfo,resultOut);
00086                 //revert back
00087                 colObj->setCollisionShape( orgShape);
00088                 colObj->setWorldTransform(  orgTrans );
00089         }
00090 }
